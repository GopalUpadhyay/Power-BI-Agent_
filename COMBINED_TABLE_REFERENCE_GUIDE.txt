QUERY STRUCTURE:
═════════════════════════════════════════════════════════════════════════

SELECT 
    sp.EmployeeID as SalesPersonID,
    sp.Name as SalesPersonName,
    sp.RegionID,
    r.RegionName,
    s.ProductID,
    p.ProductName,
    s.Amount as SalesAmount,
    s.Date as TransactionDate,
    t.TargetAmount,
    t.TargetDate,
    ROUND((s.Amount / t.TargetAmount) * 100, 2) as AchievementPercentage,
    CASE 
        WHEN s.Amount >= t.TargetAmount THEN 'Exceeded'
        WHEN s.Amount >= (t.TargetAmount * 0.9) THEN 'Met'
        ELSE 'Below'
    END as StatusVsTarget

FROM Sales s
INNER JOIN SalesPerson sp ON s.EmployeeID = sp.EmployeeID
INNER JOIN Product p ON s.ProductID = p.ProductID
INNER JOIN Region r ON sp.RegionID = r.RegionID
LEFT JOIN Targets t ON sp.EmployeeID = t.EmployeeID 
    AND MONTH(s.Date) = MONTH(t.TargetDate)
    AND YEAR(s.Date) = YEAR(t.TargetDate)

ORDER BY s.Date DESC, sp.Name;
